## Installation

```bash
npx shadcn-vue@latest add "https://nxui.dev/r/collection-surfer.json"
```

## Usage

```vue
<script setup lang="ts">
  import CollectionSurfer from '~/components/ui/CollectionSurfer.vue';
  import type { CollectionItem } from '~/types/components';

  const items: CollectionItem[] = [
    {
      id: '1',
      title: 'Mountain Escape',
      description: 'A serene mountain landscape',
      color: 'linear-gradient(135deg, #667eea 0%, #764ba2 100%)',
    },
    {
      id: '2',
      title: 'Ocean Breeze',
      description: 'Calm ocean waves at sunset',
      color: 'linear-gradient(135deg, #f093fb 0%, #f5576c 100%)',
    },
  ];
</script>

<template>
  <CollectionSurfer :items="items" class="h-[350px] w-full" />
</template>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `items` | `CollectionItem[]` | `[]` | Array of collection items to display |
| `class` | `string` | `''` | Additional CSS classes for styling |
