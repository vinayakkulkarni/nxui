## Installation

```bash
npx shadcn-vue@latest add "https://nxui.dev/r/magnetic-dock.json"
```

## Usage

```vue
<script setup lang="ts">
  import MagneticDock from '~/components/ui/MagneticDock.vue';
  import type { DockItem } from '~/types/components';

  const items: DockItem[] = [
    { icon: 'lucide:home', label: 'Home' },
    { icon: 'lucide:search', label: 'Search' },
    { icon: 'lucide:mail', label: 'Mail' },
    { icon: 'lucide:settings', label: 'Settings' },
  ];
</script>

<template>
  <MagneticDock :items="items" />
</template>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `items` | `DockItem[]` | `[]` | Array of dock items with icon and label |
| `magnification` | `number` | `1.5` | Magnification factor on hover |
| `distance` | `number` | `150` | Distance for magnetic effect in pixels |
| `class` | `string` | `''` | Additional CSS classes for styling |
