## Installation

```bash
npx shadcn-vue@latest add "https://nxui.dev/r/flight-status-card.json"
```

## Usage

```vue
<script setup lang="ts">
  import FlightStatusCard from '~/components/ui/FlightStatusCard.vue';
  import type { FlightInfo } from '~/types/components';

  const flight: FlightInfo = {
    airline: 'United Airlines',
    flightNumber: 'UA 2471',
    departure: { city: 'San Francisco', code: 'SFO', time: '08:30 AM' },
    arrival: { city: 'New York', code: 'JFK', time: '05:15 PM' },
    status: 'in-flight',
    gate: 'B42',
    progress: 65,
  };
</script>

<template>
  <FlightStatusCard :flight="flight" />
</template>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `flight` | `FlightInfo` | required | Flight information object with airline, route, and status |
| `class` | `string` | `''` | Additional CSS classes for styling |
