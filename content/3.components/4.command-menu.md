## Installation

```bash
npx shadcn-vue@latest add "https://nxui.dev/r/command-menu.json"
```

## Usage

```vue
<script setup lang="ts">
  import { ref } from 'vue';
  import CommandMenu from '~/components/ui/CommandMenu.vue';
  import type { CommandItem } from '~/types/components';

  const open = ref(false);
  const items: CommandItem[] = [
    { label: 'Home', icon: 'lucide:home', shortcut: '⌘H', group: 'Navigation' },
    { label: 'Settings', icon: 'lucide:settings', shortcut: '⌘,', group: 'Navigation' },
    { label: 'Search', icon: 'lucide:search', shortcut: '⌘F', group: 'Actions' },
  ];
</script>

<template>
  <button @click="open = true">
    Open Command Menu
  </button>
  <CommandMenu v-model:open="open" :items="items" />
</template>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `items` | `CommandItem[]` | `[]` | Array of command items to display |
| `placeholder` | `string` | `'Type a command or search...'` | Placeholder text for search input |
| `open` | `boolean` | `false` | Controls menu visibility via v-model |
| `class` | `string` | `''` | Additional CSS classes for styling |
